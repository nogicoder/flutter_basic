Widget getLeftIcon() {
  return widget.leftIcon != null
      ? Padding(
          padding: EdgeInsets.only(right: 8.0),
          child: Icon(
            widget.leftIcon,
            size: 18.0,
            color: _isPressing
                ? buttonTheme.highLightTextColor
                : buttonTheme.textColor,
          ),
        )
      : Container();
}
