UIKitDatePicker constructor

UIKitDatePicker({Key key, @required DateTime selectedDate, @required ValueChanged<DateTime> onChanged, @required DateTime firstDate, @required DateTime lastDate, SelectableDayPredicate selectableDayPredicate, DragStartBehavior dragStartBehavior: DragStartBehavior.start, bool isVertical: false })

Creates a month picker.

Rarely used directly. Instead, typically used as part of the dialog shown by showDatePicker.

Implementation

UIKitDatePicker(
    {Key key,
    @required this.selectedDate,
    @required this.onChanged,
    @required this.firstDate,
    @required this.lastDate,
    this.selectableDayPredicate,
    this.dragStartBehavior = DragStartBehavior.start,
    this.isVertical = false})
    : assert(selectedDate != null),
      assert(onChanged != null),
      assert(!firstDate.isAfter(lastDate)),
      assert((selectedDate
                  .isAfter(DateTime(firstDate.year, firstDate.month, 1)) ||
              selectedDate.isAtSameMomentAs(
                  DateTime(firstDate.year, firstDate.month, 1))) &&
          (selectedDate.isBefore(
              DateTime(lastDate.year, lastDate.month + 1 % 12, 1)))),
      super(key: key);
