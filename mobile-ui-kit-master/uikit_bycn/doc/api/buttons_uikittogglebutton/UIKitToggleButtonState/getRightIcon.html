Widget getRightIcon() {
  return widget.rightIcon != null
      ? Padding(
          padding: EdgeInsets.only(left: 8.0),
          child: Icon(
            widget.rightIcon,
            size: 18.0,
            color: _isActivated
                ? buttonTheme.activatedTextColor
                : buttonTheme.textColor,
          ),
        )
      : Container();
}
